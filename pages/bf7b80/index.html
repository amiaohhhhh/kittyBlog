    dataPicker限制选择日期范围

    # el-dataPicker 限制选择日期范围

    需要实现一个日期选择功能，能够输入开始日期 结束日期，展示格式为yyyy-MM-dd，整个数值返回为一个数值（例如：[2022-08-24,2020-08-30]）
    具体效果
    只能选半年内的 当天不能选

    要快捷键
    时间跨度一个月的范围 当天不能选

    # 实现代码

    <el-date-picker size="mini"
 	v-model="searchTime"
  :clearable = 'false'
  type="daterange"
  align="center"
  :picker-options="pickerOptions"
  unlink-panels
  range-separator="至"
  start-placeholder="开始日期"
  end-placeholder="结束日期"
  format='yyyy-MM-dd' value-format='yyyy-MM-dd'
  style='width:260px;margin-right:20px'
  >
</el-date-picker>

    export default {
  data(){
    let _minTime = null
    let _maxTime = null    
    return{
      searchTime:'',
      pickerOptions: {
        onPick(time) { 
          //此处的time为选择日期时的结果，time为一个对象，有两个值，一个是maxDate,一个是minDate,详细见下图，
          //以下判断是只选择了一个值的时候，也就是默认的开始值。
          if (!time.maxDate) { 
          //时间跨度为1天，1天转化成毫秒就是 24 * 3600 * 1000
            let timeRange = 29 * 24 * 60 * 60 * 1000
            //最小值为选择的日期-跨度值
            _minTime = time.minDate.getTime() - timeRange
            //最大值为选择的日期+跨度值
            _maxTime = time.minDate.getTime() + timeRange
          } else { 
          //如果两个值都已经选择了，也就是已经选择完成，则将最大值和最小值设置为null
            _minTime = _maxTime = null
          }
        },
        //当已经选择了两个日期时
        disabledDate(time) { 
          if (_minTime && _maxTime) { 
          //如果已经选择了两个日期，则小于最小值和大于最大值的都要禁用和当天
            return time.getTime() < _minTime || time.getTime() > _maxTime || time.getTime() > Date.now()- 8.64e7
          } else { 
          //如果没有选择两个值，即选择了一个或者什么都没有选择的初始状态，禁用的日期是当前日期之前的日期和当天
            return time.getTime() < (Date.now() -180 * 24 * 60 * 60 * 1000 ) || time.getTime() > Date.now()- 8.64e7
            }
        },
        //快捷键
        shortcuts: [{
          text: '昨天',
          onClick(picker) {
            const end = new Date();
            const start = new Date();
            start.setTime(start.getTime() - 3600 * 1000 * 24 * 1);
            end.setTime(end.getTime() - 3600 * 1000 * 24 * 1);
            picker.$emit('pick', [start, end]);
            }
      	},{
            text: '最近三天',
            onClick(picker) {
            const end = new Date();
            const start = new Date();
            start.setTime(start.getTime() - 3600 * 1000 * 24 * 2);
            picker.$emit('pick', [start, end]);
            }
      	},{
            text: '最近一周',
            onClick(picker) {
            const end = new Date();
            const start = new Date();
            start.setTime(start.getTime() - 3600 * 1000 * 24 * 6);
            picker.$emit('pick', [start, end]);
            }
      	}, {
            text: '最近二周',
            onClick(picker) {
            const end = new Date();
            const start = new Date();
            start.setTime(start.getTime() - 3600 * 1000 * 24 * 13);
            picker.$emit('pick', [start, end]);
            },
      	},{
            text: '最近一个月',
            onClick(picker) {
            const end = new Date();
            const start = new Date();
            start.setTime(start.getTime() - 3600 * 1000 * 24 * 29);
            picker.$emit('pick', [start, end]);
            },
        }],
      },
