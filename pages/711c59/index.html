mounted(){
  if(this.$refs.upload){
    this.$refs.upload.addEventListener('change', e => {//绑定监听表格导入事件
      this.inport_excel(e)
    })
  }
},
methods:{
  async inport_excel(e) {//表格导入
    let that = this
    const files = e.target.files;
    if(files.length<=0){//如果没有文件名
      return false;
    }else if(!/\.(xls|xlsx)$/.test(files[0].name.toLowerCase())){
      this.$message.error('上传格式不正确，请上传xls或者xlsx格式');
      return false;
    }
    const fileReader = new FileReader();
    fileReader.onload = async (ev) => {
      const data = ev.target.result;
      const workbook = XLSX.read(data, {
          type: 'binary'
      });
      const wsname = workbook.SheetNames[0];//取第一张表
      const ws = XLSX.utils.sheet_to_json(workbook.Sheets[wsname]);//生成json表格内容
      that.outputsList = [];//清空接收数据
      // 数据处理
      this.outputsList = JSON.parse(JSON.stringify(ws)) //打印出除了表名的内容
      this.$refs.upload.value = '';
    };
    fileReader.readAsBinaryString(files[0]);
  },
}

<script>
  import XLSX from 'xlsx';//引入xlsx
  export default {
  data(){
    return{
      outputsList:[],//表格
      outputHeadList:[],//表头
    }
  },
  mounted(){
    // 监听 点击按钮后 弹起选择文件框
    if(this.$refs.upload){
      this.$refs.upload.addEventListener('change', e => {//绑定监听表格导入事件
        this.inport_excel(e)
      })
    }
  },
  methods:{
    choose_File(){
      this.$refs.upload.click() //启动文件框(相当于点了input一下)
      this.stauts = 1
    },
    // 点击导入
    inport_excel(e){
      let that = this
      const files = e.target.files;
      if(files.length<=0){//如果没有文件名
        this.$message.error('没有上传文件');
        that.$refs.upload.value = ''
        return false;
      }else if(!/\.(xls|xlsx)$/.test(files[0].name.toLowerCase())){
        this.$message.error('上传格式不正确，请上传xls或者xlsx格式');
        that.$refs.upload.value = ''
        return false;
      }
      const fileReader = new FileReader();
      fileReader.onload = async (ev) => {
        const data = ev.target.result;
        const workbook = XLSX.read(data, {
          type: 'binary',
          cellDates:true
        });
        const wsname = workbook.SheetNames[0];//取第一张表表名
        let workbookSheet =  workbook.Sheets[wsname]
        const ws = XLSX.utils.sheet_to_json(workbookSheet);//生成json表格内容
        that.outputsList = [];//清空接收数据
        if(ws.length == 0){
          this.$message.error('导入的表中没有数据')
          that.$refs.upload.value = '';
          return
        }
        this.isLoading = true
        //获取表头
        let headNum = workbookSheet['!ref'] 
        //得到的是'A1:C9' 把两个字母给拿出来 就可以知道表头的范围了
        let startEnd = headNum.substring(headNum.indexOf(':') + 1,headNum.indexOf(':') + 2)
        let i = 65
        let arr = []
        // String.fromCharCode：将 Unicode 编码转换为一个字符串
        // 才能循环批量拿到数组中的表头名
        while(String.fromCharCode(i) <= startEnd){
          //提取表头
          let list = workbookSheet[String.fromCharCode(i) + '1']
          arr.push(list.v)
          i++
        }
        //批量添加 ws提取出来的表数据 没有表头
        this.outputsList = JSON.parse(JSON.stringify(ws))
        this.outputHeadList = arr
      	// 清空input框数据
        that.$refs.upload.value = '';
        console.log(outputsList)
      };
      fileReader.readAsBinaryString(files[0]);
    },
  }
}
</script>

// 导出文件 方法
export_excel(){
  let fs = require('fs');
  let os = require('os');
  let homedir = os.homedir();
  this.limlt = this.total
  let outputsList = JSON.parse(JSON.stringify(this.successList))
  let arr=outputsList.map((v,i)=>{
      return {
          '确认状态':v.status==1?'是':'否',
          '来源':v.source,
          '日期':v.date
      }
  })
  //文件名
  let fileName = '成功案例' 
  var worksheet = this.$xlsx.utils.json_to_sheet(arr);
  var new_workbook = this.$xlsx.utils.book_new();
  //实现多个子表导出
  //下面这句话很关键
  this.$xlsx.utils.book_append_sheet(new_workbook, worksheet, "sheetjs");
  // 如果多个子表导出 即循环上句 把'sheet.js'改成该子表名
  // 判断有没有这个文件 手动帮他保存到指定路径 （桌面）
  let filePath = homedir + "\\Desktop"
  let that = this
  fs.exists(filePath, function(exists) {
    if(exists){
      that.$message.success('导出成功,下载图片中')
      that.$xlsx.writeFile(new_workbook, homedir + "\\Desktop\\" + fileName + ".xlsx");
    }else{
      that.$message.error('请到D盘\成功案例中获取文件')
      that.$xlsx.writeFile(new_workbook,"D\\" + fileName + ".xlsx");
    }
  });
},
