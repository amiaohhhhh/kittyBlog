    # 【JS】分片上传视频

    需求分析:上传视频文件并以1M分片的形式上传，且在线浏览

    # 后台文档

    let formData = new FormData()
formData.append('bucket', 'xyzyw') // 保存视频的容器
formData.append('object', this[type].video_object) // 视频的文件名
formData.append('position', this[type].video_position) // 切片下标
formData.append('file', chunk[0], this[type].video_fileName) // 切片源
this.$http.post('//api.ysb.cn/oss/Upload/appendFrom', formData,
	{headers: {'content-type': 'multipart/form-data'}}
).then(res => {
	if (res.data.code == 1) {
		this[type].video_position = res.data.data // 修改切片下标
		this[type].video_chunk    = chunk.slice(1) // 删除刚上传的切片
		this.videoSliceUpload(type) // 切片上传
	} else {
		this.setProgress(false, 0) // 还原进度条
		this.$message.error('上传视频失败')
	}
})


视频域名：https://image.ysb.cn/
视频路径：sns/video/
完整的视频路径：https://image.ysb.cn/sns/video/视频名称.mp4

    # 前端代码

    <template>
    <div id="app">
        <div class="album albumvideo">
            <div>
                <p class="type_title" style="margin:100px auto;">
                    <span>上传视频到服务器</span>
                </p>
                <div class="pic_img">
                    <div class="pic_img_box">
                        <el-upload class="avatar-uploader" action="#" :http-request='aaafile' :on-progress="uploadVideoProcess" :on-success="handleVideoSuccess" :before-upload="beforeUploadVideo" :show-file-list="true" >
                            <video v-if="videoForm.showVideoPath !='' && !videoFlag" v-bind:src="videoForm.showVideoPath" class="avatar video-avatar" controls="controls">
                                您的浏览器不支持视频播放
                            </video>
                            <i v-else-if="videoForm.showVideoPath =='' && !videoFlag" >
                                <el-button slot="trigger" size="small" type="primary">选取文件上传</el-button>
                            </i>
                            <el-progress v-if="videoFlag == true" type="circle" v-bind:percentage="videoUploadPercent" style="margin-top:7px; "></el-progress>
                        </el-upload>
                    </div>
                </div>
            </div>
        </div>
    </div>
</template>
<style lang="css" scoped>
video{
    width: 300px;
}
</style>
<script>
// import OSS from 'ali-oss'
// let ossConfig = {
//     accessKeyId: "yourid",
//     accessKeySecret: "yourkey",
//     bucket: "**",
//     region: "**"
// };
    export default {
        data() {
            return {
                smdx:null,
                videoFlag: false,
                //是否显示进度条
                videoUploadPercent: "",
                //进度条的进度，
                isShowUploadVideo: false,
                //显示上传按钮
                videoForm: {
                    showVideoPath: ''
                },
                //创建一个字段
                aa:0
            }
        },
        methods: {
            async aaafile(e){
                var d1 = new Date();
                let file = this.smdx; //上传文件主体
                console.log(file)
                let name = 'sns/video/'+file.name, //文件名 加上时间戳
                size = file.size, //总大小
                succeed = 0, //当前上传数
                shardSize = 1 * 1024 * 1024, //以1MB为一个分片
                shardCount = Math.ceil(size / shardSize); //总片数

                let attr = [];
                try {
                    for (let item = 0; item < shardCount; ++item) {
                        await this.fn(item,shardSize,size,name,shardCount,file,succeed); //同步
                        // attr.push(fn(item)); //异步
                    }
                    await Promise.all(attr); //异步
                    console.log(' 上传成功');
                    var d2 = new Date();
                    console.log(parseInt(d2 - d1) / 1000);
                } catch (err) {
                    console.log(err);
                }
            },
            fn(item,shardSize,size,name,shardCount,file,succeed){
                 return new Promise((resolve, reject) => {
                    var i = item;
                    console.log(i);//分片下标
                    var start = i * shardSize, //当前分片开始下标
                    end = Math.min(size, start + shardSize); //结束下标

                    //构造一个表单，FormData是HTML5新增的
                    var formData = new FormData();
                    formData.append('bucket', 'xyzyw') // 保存视频的容器
                    formData.append("file", file.slice(start, end) , name.split(',')[0]); //slice方法用于切出文件的一部分
                    formData.append("object", name); //文件名字
                    // formData.append("total", shardCount); //总片数
                    formData.append("position", this.aa); //当前片数 回调上一个的值
                    //Ajax提交
                    let config = {
                    headers: { "Content-Type": "multipart/form-data" }
                     };
                    this.$axios.post( "http://api.ysb.cn/oss/Upload/appendFrom",formData,config
                    ).then(data=>{
                            ++succeed;
                            if (typeof (data) == 'string') {
                                try {
                                    data = JSON.parse(data);
                                    console.log(data.msg);
                                } catch (e) {
                                    console.log(data);
                                }
                            }
                            //生成当前进度百分比
                            var jd = `${Math.round(succeed / shardCount * 100)}%`;
                            console.log(jd);
                            /*如果是线上，去掉定时，直接callback()，
                            这样写是为方便，本地测试看到进度条变化
                            因为本地做上传测试是秒传，没有时间等待*/
                            this.aa = data.data.data;
                            setTimeout(resolve, 50);
                    })
                })
            },
            //上传前回调
            beforeUploadVideo(file) {
                console.log(file)
                var fileSize = file.size / 1024 / 1024 < 500;
                if (['video/mp4', 'video/ogg', 'video/flv', 'video/avi', 'video/wmv', 'video/rmvb', 'video/mov'].indexOf(file.type) == -1) {
                    layer.msg("请上传正确的视频格式");
                    return false;
                }
                if (!fileSize) {
                    layer.msg("视频大小不能超过500MB");
                    return false;
                }
                this.smdx = file;
                this.isShowUploadVideo = false;
            },
            //回调上一个视频信息
            //进度条
            uploadVideoProcess(event, file, fileList) {
                this.videoFlag = true;
                this.videoUploadPercent = file.percentage.toFixed(0) * 1;
            },
            //上传成功回调
            handleVideoSuccess(res, file) {
                this.isShowUploadVideo = true;
                this.videoFlag = false;
                this.videoUploadPercent = 0;
                
                // 前台上传地址
                if (file.status == 'success' ) {
                    this.videoForm.showVideoPath = 'https://image.ysb.cn/sns/video/'+file.name;
                    alert("上传成功")
                } else {
                    alert("上传失败，请重新上传");
                }

                //后台上传地址
                // if (res.data.code == 0) {
                //     this.videoForm.showVideoPath = res.Data;
                // } else {
                //     layer.msg(res.Message);
                // }
                // if(res.data.code == 1){
                //     this.i = res.data.data
                //     this.aaafile(file)
                // }else{
                //     layer.msg(res.Message)
                // }
            }
        }
    }
</script>
