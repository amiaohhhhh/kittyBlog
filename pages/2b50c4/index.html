    时间日期选择器自定义日期精确到秒

    # 时间日期选择器自定义日期精确到秒

    # 需求

    点击今天起始的时间是从00:00:00开始--到所点击时间结束

    # 效果

    # 解决方法

    <div>
 	<el-date-picker 
  	v-model="time" 
    type="datetimerange" 
    range-separator="至" 
    start-placeholder="开始日期" 
    end-placeholder="结束日期" 
    size="mini"
    value-format='yyyy-MM-dd HH:mm:ss' 
    :default-time="['00:00:00','23:59:59']"  
    :picker-options="pickerOption"></el-date-picker>
</div>

    //丢到data里面
export default {
  data(){
    return{
    	pickerOption: {
        disabledDate(time) {
            return time.getTime() > Date.now() ;
        },
        shortcuts: [{
            text: '今天',
            onClick(picker) {
            const end = new Date();
            let a = new Date().toLocaleDateString() + ' 00:00:00'
            const start = new Date(Date.parse(a));
            picker.$emit('pick', [start, end]);
          	}
        },{
          text: '昨天',
          onClick(picker) {
            const a = new Date().toLocaleDateString() + ' 00:00:00'
            const b = new Date().toLocaleDateString() + ' 23:59:59'
            const end = new Date(Date.parse(b)- 3600 * 1000 * 24 * 1);
            const start = new Date(Date.parse(a)- 3600 * 1000 * 24 * 1);
            picker.$emit('pick', [start, end]);
          }
        },
      }
    }
  }
}
